83 torch::Tensor boundary_loss(PINN& model) {
84 auto options = torch::TensorOptions().dtype(torch::kFloat32).requires_grad(true);
85
86 // x = 0 point
87 auto x0 = torch::zeros({ 1, 1 }, options).set_requires_grad(true);
88 auto u0 = model.forward(x0);
89 auto du0 = torch::autograd::grad({ u0 }, { x0 }, { torch::ones_like(u0) },
90 Global::keep_graph, true)[0];
91
92 // x = 1 point
93 auto x1 = torch::ones({ 1, 1 }, options).set_requires_grad(true);
94 auto u1 = model.forward(x1);
95
96 auto du1 = torch::autograd::grad({ u1 }, { x1 }, { torch::ones_like(u1) },
97 Global::keep_graph, true)[0];
98 auto d2u1 = torch::autograd::grad({ du1 }, { x1 }, { torch::ones_like(du1) },
99 Global::keep_graph, true)[0];
100 auto d3u1 = torch::autograd::grad({ d2u1 }, { x1 }, { torch::ones_like(d2u1) },
101 Global::keep_graph, true)[0];
102
103 // Sum of MSE losses for each boundary condition
104 auto loss = torch::mse_loss(u0, torch::zeros_like(u0)) +
105 torch::mse_loss(du0, torch::zeros_like(du0)) +
106 torch::mse_loss(d2u1, torch::zeros_like(d2u1)) +
107 torch::mse_loss(d3u1, torch::zeros_like(d3u1));
108
109 return loss;
110 }

124 torch::Tensor physics_loss(PINN& model, torch::Tensor input, float EI = 1.0f) {
125 try {
126 auto x = input.clone().requires_grad_(true);
127 auto u = model.forward(x);
128 auto ones = torch::ones_like(u);
129
130 auto du_dx = torch::autograd::grad({ u }, { x }, { ones }, Global::keep_graph, true)[0];
131 auto d2u_dx2 = torch::autograd::grad({ du_dx }, { x }, { ones }, Global::keep_graph, true)[0];
132 auto d3u_dx3 = torch::autograd::grad({ d2u_dx2 }, { x }, { ones }, Global::keep_graph, true)[0];
133 auto d4u_dx4 = torch::autograd::grad({ d3u_dx3 }, { x }, { ones }, Global::keep_graph, true)[0];
134
135 auto q = torch::ones_like(x); // Constant load q(x) = 1
136
137 auto residual = EI * d4u_dx4 - q;
138
139 return torch::mean(residual.pow(2));
140 }
141 catch (const std::exception& e) {
142 std::cerr << "Error in physics_loss: " << e.what() << std::endl;
143 return torch::tensor(0.0f, torch::requires_grad(true));
144 }
145 }

154 torch::Tensor compute_l2_regularization(PINN& model, const float lambda_reg) {
155 if (lambda_reg <= 0.0f) {
156 return torch::zeros({ 1 }, torch::TensorOptions().dtype(torch::kFloat32));
157 }
158
159 torch::Tensor l2 = torch::zeros({ 1 }, torch::TensorOptions().dtype(torch::kFloat32));
160 for (const auto& param : model.parameters()) {
161 l2 += torch::norm(param, 2).pow(2);
162 }
163
164 return lambda_reg * l2;
165 }

176 Losses compute_losses(PINN& model, const torch::Tensor& physics_input) {
177 try {
178 auto loss_physics = physics_loss(model, physics_input);
179 auto loss_boundary = boundary_loss(model);
180 auto l2_reg = compute_l2_regularization(model, 0.0f);
181
182 auto total = loss_physics + 2.0f * loss_boundary + l2_reg;
183
184 return { total, loss_physics, loss_boundary, l2_reg };
185 }
186 catch (const std::exception& e) {
187 std::cerr << "Error in compute_losses: " << e.what() << std::endl;
188 auto dummy = torch::tensor(1.0f, torch::requires_grad(true));
189 return { dummy, dummy, dummy, dummy };
190 }
